✅ Controller:

use Illuminate\Support\Facades\Artisan;

if($request->clear_cache_item == 'Yes') {
    Artisan::call('cache:clear');
    Artisan::call('config:clear');
    Artisan::call('view:clear');
    Artisan::call('route:clear');
    Artisan::call('optimize:clear');
}
